```python
import os
import shutil
import json
from pathlib import Path
from datetime import datetime


RAW_DIR = Path("01_RAW_ARCHIVE")
CUSTOM_DIR = Path("02_CUSTOM")


# -------------------------
# UTILS
# -------------------------

def sanitize_name(name: str) -> str:
    return name.strip().replace(" ", "_")


def ensure_dir(path: Path):
    path.mkdir(parents=True, exist_ok=True)


# -------------------------
# PROCESS PARSER (BASE)
# -------------------------

def parse_process_description(text: str) -> dict:
    text = text.lower()

    process = "unknown"
    tool = "unknown"
    generated_from = []

    if "normal" in text and "albedo" in text:
        process = "normal_from_albedo"
        generated_from.append("albedo")

    if "materialize" in text:
        tool = "materialize"

    if "substance" in text:
        tool = "substance_designer"

    return {
        "process": process,
        "tool": tool,
        "generated_from": generated_from,
        "confidence": 0.7
    }


# -------------------------
# CORE IMPORT
# -------------------------

def import_asset(
    source_path: str,
    asset_type: str,
    provider: str,
    material_name: str,
    process_description: str = ""
):
    """
    asset_type: "RAW" | "CUSTOM"
    """

    source = Path(source_path)

    if not source.exists():
        raise FileNotFoundError(f"Source not found: {source}")

    provider = sanitize_name(provider)
    material_name = sanitize_name(material_name)

    # -------------------------
    # DESTINATION
    # -------------------------
    if asset_type.upper() == "RAW":
        dest_root = RAW_DIR / provider / material_name
    elif asset_type.upper() == "CUSTOM":
        dest_root = CUSTOM_DIR / provider / material_name
    else:
        raise ValueError("asset_type must be RAW or CUSTOM")

    ensure_dir(dest_root)

    # -------------------------
    # COPY FILES
    # -------------------------
    copied_files = []

    if source.is_file():
        dest = dest_root / source.name
        shutil.copy2(source, dest)
        copied_files.append(dest.name)

    elif source.is_dir():
        for root, _, files in os.walk(source):
            for f in files:
                src_file = Path(root) / f
                rel_path = Path(root).relative_to(source)
                dest_folder = dest_root / rel_path
                ensure_dir(dest_folder)

                dest_file = dest_folder / f
                shutil.copy2(src_file, dest_file)
                copied_files.append(str(dest_file.relative_to(dest_root)))

    # -------------------------
    # MATERIAL INFO
    # -------------------------
    material_info = {
        "material": material_name,
        "provider": provider,
        "type": asset_type.upper(),
        "import_date": datetime.now().isoformat(),
        "files": copied_files,
        "notes": ""
    }

    with open(dest_root / "material_info.json", "w", encoding="utf-8") as f:
        json.dump(material_info, f, indent=2)

    # -------------------------
    # PROCESS (solo custom)
    # -------------------------
    if asset_type.upper() == "CUSTOM":
        process_data = parse_process_description(process_description)

        process_json = {
            "source": "custom",
            "pipeline": [
                {
                    "output": "generated_maps",
                    "tool": process_data["tool"],
                    "process": process_data["process"],
                    "generated_from": process_data["generated_from"],
                    "confidence": process_data["confidence"]
                }
            ]
        }

        with open(dest_root / "process.json", "w", encoding="utf-8") as f:
            json.dump(process_json, f, indent=2)

    print(f"\n✔ Imported: {material_name}")
    print(f"   → {dest_root}")
    print(f"   Files: {len(copied_files)}\n")


# -------------------------
# CLI TEST
# -------------------------

if __name__ == "__main__":
    # ESEMPIO USO

    import_asset(
        source_path="path/to/your/files",
        asset_type="RAW",
        provider="AmbientCG",
        material_name="Tiles_Example"
    )

    import_asset(
        source_path="path/to/custom/files",
        asset_type="CUSTOM",
        provider="Custom",
        material_name="Tiles_Example_Custom",
        process_description="normal da albedo con materialize"
    )
```

# 🔥 CODICE IMPORTER (BASE SERIA, NON GIOCATTOLINO)

```python
import os
import shutil
import json
import requests
from pathlib import Path

# -------------------------
# CONFIG
# -------------------------

OLLAMA_URL = "http://localhost:11434/api/generate"
OLLAMA_MODEL = "mistral"  # cambia se vuoi (llama3 ecc)

RAW_DIR = Path("01_RAW_ARCHIVE")
CUSTOM_DIR = Path("02_CUSTOM")
TEMP_DIR = Path("temp/import_batch")

VALID_EXT = [".png", ".jpg", ".jpeg", ".tif", ".tiff", ".exr"]

# -------------------------
# AI CALL (OLLAMA)
# -------------------------

def ask_ai(prompt):
    try:
        response = requests.post(
            OLLAMA_URL,
            json={
                "model": OLLAMA_MODEL,
                "prompt": prompt,
                "stream": False
            }
        )
        return response.json()["response"]
    except Exception as e:
        print("AI ERROR:", e)
        return None

# -------------------------
# USER INPUT
# -------------------------

def ask_user(prompt):
    return input(f"\n{prompt}\n> ").strip()

# -------------------------
# CLASSIFICATION
# -------------------------

def classify_file(file_path):
    name = file_path.name.lower()

    # euristica base
    if any(x in name for x in ["normal", "rough", "albedo", "ao"]):
        return "raw_candidate"

    return "unknown"

# -------------------------
# AI PROCESS INTERPRETER
# -------------------------

def interpret_process(description):
    prompt = f"""
Convert this description into structured JSON pipeline.

Description:
{description}

Output JSON format:
{{
  "pipeline": [
    {{
      "step": "...",
      "tool": "...",
      "input": "...",
      "output": "...",
      "notes": "..."
    }}
  ]
}}
"""

    result = ask_ai(prompt)

    try:
        return json.loads(result)
    except:
        return {"raw_text": description}

# -------------------------
# IMPORT RAW
# -------------------------

def import_raw(files, provider_name, material_name):

    target = RAW_DIR / provider_name / material_name
    target.mkdir(parents=True, exist_ok=True)

    for f in files:
        shutil.copy2(f, target / f.name)

    print(f"[RAW] Imported → {target}")

# -------------------------
# IMPORT CUSTOM
# -------------------------

def import_custom(files, material_name, process_data):

    target = CUSTOM_DIR / material_name
    target.mkdir(parents=True, exist_ok=True)

    for f in files:
        shutil.copy2(f, target / f.name)

    # salva process.json
    with open(target / "process.json", "w", encoding="utf-8") as fp:
        json.dump(process_data, fp, indent=2)

    print(f"[CUSTOM] Imported → {target}")

# -------------------------
# MAIN IMPORT FLOW
# -------------------------

def run_importer():

    if not TEMP_DIR.exists():
        print("Temp folder not found")
        return

    files = [f for f in TEMP_DIR.iterdir() if f.suffix.lower() in VALID_EXT]

    if not files:
        print("No valid files found")
        return

    print(f"\nFound {len(files)} files")

    mode = ask_user("Are these RAW or CUSTOM? (r/c)")

    if mode == "r":
        provider = ask_user("Provider name? (e.g. AmbientCG)")
        material = ask_user("Material name?")

        import_raw(files, provider, material)

    elif mode == "c":
        material = ask_user("Base material name?")
        desc = ask_user("Describe the process used (free text)")

        process_data = interpret_process(desc)

        import_custom(files, material, process_data)

    else:
        print("Invalid mode")

# -------------------------
# ENTRY
# -------------------------

if __name__ == "__main__":
    run_importer()
```

import os
import shutil
import json
import requests
from pathlib import Path

# -------------------------
# CONFIG
# -------------------------

OLLAMA_URL = "http://localhost:11434/api/generate"
OLLAMA_MODEL = "mistral"

RAW_DIR = Path("01_RAW_ARCHIVE")
CUSTOM_DIR = Path("02_CUSTOM")
TEMP_DIR = Path("temp/import_batch")

VALID_EXT = [".png", ".jpg", ".jpeg", ".tif", ".tiff", ".exr"]

# -------------------------
# AI
# -------------------------

def ask_ai(prompt):
    try:
        r = requests.post(
            OLLAMA_URL,
            json={"model": OLLAMA_MODEL, "prompt": prompt, "stream": False}
        )
        return r.json()["response"]
    except:
        return None

def interpret_process(description):
    prompt = f"""
Convert this into structured JSON pipeline.

{description}

Output JSON only:
{{
 "pipeline":[{{"step":"","tool":"","input":"","output":"","notes":""}}]
}}
"""
    result = ask_ai(prompt)

    try:
        return json.loads(result)
    except:
        return {"raw_text": description}

# -------------------------
# USER INPUT HELPERS
# -------------------------

def ask(msg):
    return input(f"\n{msg}\n> ").strip()

def choose(options):
    for i, opt in enumerate(options):
        print(f"{i}: {opt}")
    idx = int(input("> "))
    return options[idx]

# -------------------------
# SESSION DATA
# -------------------------

materials = {}

def get_or_create_material():
    name = ask("Material name?")

    if name not in materials:
        provider = ask("Provider? (es: AmbientCG, Custom, Poliigon)")
        materials[name] = {
            "provider": provider,
            "raw_files": [],
            "custom_files": [],
            "process": None
        }

    return name

# -------------------------
# FILE ASSIGNMENT
# -------------------------

def process_files(files):

    for f in files:

        print("\n" + "="*50)
        print(f"FILE: {f.name}")
        print("="*50)

        action = ask(
            "Action?\n"
            "1 = RAW\n"
            "2 = CUSTOM\n"
            "3 = SKIP"
        )

        if action == "3":
            continue

        mat_name = get_or_create_material()

        if action == "1":
            materials[mat_name]["raw_files"].append(f)

        elif action == "2":
            materials[mat_name]["custom_files"].append(f)

# -------------------------
# SAVE RAW
# -------------------------

def save_raw():

    for mat, data in materials.items():

        if not data["raw_files"]:
            continue

        target = RAW_DIR / data["provider"] / mat
        target.mkdir(parents=True, exist_ok=True)

        for f in data["raw_files"]:
            shutil.copy2(f, target / f.name)

        print(f"[RAW] {mat} → {len(data['raw_files'])} files")

# -------------------------
# SAVE CUSTOM
# -------------------------

def save_custom():

    for mat, data in materials.items():

        if not data["custom_files"]:
            continue

        target = CUSTOM_DIR / mat
        target.mkdir(parents=True, exist_ok=True)

        for f in data["custom_files"]:
            shutil.copy2(f, target / f.name)

        # processo custom
        desc = ask(f"Describe CUSTOM process for {mat} (or leave empty)")

        if desc:
            process_data = interpret_process(desc)

            with open(target / "process.json", "w", encoding="utf-8") as fp:
                json.dump(process_data, fp, indent=2)

        print(f"[CUSTOM] {mat} → {len(data['custom_files'])} files")

# -------------------------
# MAIN
# -------------------------

def run_importer():

    if not TEMP_DIR.exists():
        print("Missing temp/import_batch")
        return

    files = [f for f in TEMP_DIR.iterdir() if f.suffix.lower() in VALID_EXT]

    if not files:
        print("No valid files")
        return

    print(f"\nFILES FOUND: {len(files)}")

    process_files(files)

    print("\n--- SAVING RAW ---")
    save_raw()

    print("\n--- SAVING CUSTOM ---")
    save_custom()

    print("\nIMPORT COMPLETE")

# -------------------------
# ENTRY
# -------------------------

if __name__ == "__main__":
    run_importer()
