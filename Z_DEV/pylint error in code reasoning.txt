```python
"""Metadata extraction module using exiftool and PIL."""

import json
import subprocess
from pathlib import Path
from PIL import Image

EXIFTOOL_PATH = Path("config/exiftool.exe")
VALID_EXT = {".png", ".jpg", ".jpeg", ".tif", ".tiff", ".exr"}

def extract_with_exiftool(file_path):
    """Extract metadata using exiftool."""
    try:
        result = subprocess.run(
            ["exiftool", "-j", file_path],
            capture_output=True,
            text=True,
            check=False  # ← Add explicit check parameter
        )

        if not result.stdout:
            return None

        data = json.loads(result.stdout)[0]
        return {
            "metadata_raw": data,
            "source": "exiftool"
        }

    except (json.JSONDecodeError, subprocess.SubprocessError) as e:
        return None

def extract_with_pillow(file_path: str):
    """Extract metadata using PIL as fallback."""
    try:
        with Image.open(file_path) as img:
            return {
                "resolution": {
                    "width": img.width,
                    "height": img.height
                },
                "mode": img.mode,
                "format": img.format,
                "channels": len(img.getbands()),
                "has_alpha": "A" in img.getbands(),
                "source": "pillow"
            }
    except OSError as e:
        return {"error": str(e), "source": "error"}

def extract_metadata(file_path: str) -> dict:
    """Extract metadata from image file."""
    data = extract_with_exiftool(file_path)
    if data:
        return data
    return extract_with_pillow(file_path)
```
