Ecco il protocollo v3. Questo è ciò che è cambiato sostanzialmente rispetto alla v2 e perché:

Riposizionamento fondamentale: SIGNUM SENTINEL Il documento ora parte da cosa è questo sistema realmente — non una pipeline di elaborazione texture, ma il cervello di DVAMOCLES in costruzione. Il dataset che produciamo oggi addestra Sentinel che diventa l'AI del software. Questa connessione era implicita, ora è esplicita e struttura tutto il resto.

Layer 0 — Physics Oracle (completamente nuovo) Le banche dati che hai costruito (IOR, F0, albedo ranges, SpectralDB, MaterialX) diventano un layer architetturale formale. Ogni materiale viene confrontato contro questi dati reali — non solo contro soglie generiche. Il Rame deve avere R >> B nel BaseColor. Il cemento deve avere roughness 0.80–0.95. Il legno grezzo è l'unico dielettrico con F0 > 0.04. Il Physics Oracle cresce con la tua ricerca senza toccare il codice.

Cross-Map Correlation Analyzer (nuovo Modulo 5b) Studia le correlazioni fisiche tra le mappe. Se c'è un graffio visibile nel BaseColor ma la Normal non mostra nessuna deformazione corrispondente, il sistema lo segnala e genera la decisione di propagarlo. Questa è la base tecnica per l'editing coerente che hai descritto.

Coherent Damage Propagation Non più solo "genera roughness da albedo" — ora se decido che un graffio deve esistere fisicamente, il sistema lo propaga su tutte le mappe in modo fisicamente corretto: profondità nella Normal, aumento roughness nella zona, riduzione AO ai bordi.

Scale Estimation Riconosce che un set di texture rappresenta 0.88m × 0.60m di muro a 4 file di mattoni, non solo che è "un muro di mattoni". Fondamentale per texel density e per calibrare le correzioni in unità fisiche reali.

Variant Generator deterministico Varianti broken tile, wet, dirty, aged — generate con algoritmi deterministici guidati dal Physics Oracle, non da AI generativa. Fisicamente plausibili per definizione.

Cross-Material Comparator e Blending Advisor Come si comporta la roughness su stone vs concrete. Come integrare terra + pietra in modo fisicamente corretto (height blend + AO nei bordi + contamination roughness).

# ── Folder Classification ─────────────────────────────────────────────────────

def _classify_subfolder(name: str):
    """
    Returns (kind, resolution, format).
    kind = "variant_named" | "resolution" | "format" | "unknown"
    """
    m = _VARIANT_FOLDER_RE.match(name)
    if m:
        return "variant_named", m.group(1).upper(), m.group(2).upper()
    if _RES_ONLY_RE.match(name):
        return "resolution", name.upper(), ""
    if name.lower() in _FORMAT_NAMES:
        return "format", "", name.upper()
    return "unknown", "", ""


def _is_material_folder(folder: Path) -> bool:
    """True if folder contains PBR content (images or variant subfolders)."""
    for child in folder.iterdir():
        if child.is_file():
            if child.suffix.lower() in VALID_EXTENSIONS:
                if child.stat().st_size >= MIN_FILE_SIZE:
                    if not _is_ignored(child.name):
                        return True
        elif child.is_dir():
            kind, _, _ = _classify_subfolder(child.name)
            if kind in ("variant_named", "resolution"):
                return True
    return False


def _is_provider_folder(folder: Path, root: Path) -> bool:
    """True if folder is a depth-1 provider wrapper with no direct images."""
    try:
        depth = len(folder.relative_to(root).parts)
    except ValueError:
        return False
    if depth != 1:
        return False
    name_lower = folder.name.lower()
    for key in _KNOWN_PROVIDERS:
        if key in name_lower:
            return True
    # Unknown name at depth 1 with only subdirs and no images → wrapper
    has_images = any(
        c.is_file() and c.suffix.lower() in VALID_EXTENSIONS
        for c in folder.iterdir()
    )
    if has_images:
        return False
    all_dirs = all(c.is_dir() for c in folder.iterdir() if not c.name.startswith("."))
    return all_dirs


def _is_ignored(filename: str) -> bool:
    lower = filename.lower()
    return any(p in lower for p in IGNORE_PATTERNS)


# ── Provider Detection ────────────────────────────────────────────────────────

def _detect_provider(folder: Path, root: Path):
    try:
        parts = folder.relative_to(root).parts
    except ValueError:
        return "Unknown", "UNK"
    for part in parts[:-1]:
        lower = part.lower()
        for key, (name, code) in _KNOWN_PROVIDERS.items():
            if key in lower:
                return name, code
    if len(parts) >= 2:
        ancestor = parts[0]
        code = re.sub(r'[^A-Za-z]', '', ancestor)[:3].upper() or "GEN"
        return ancestor.title(), code
    return "Unknown", "UNK"

def _collect_image_files(folder: Path) -> list:
    result = []
    for f in folder.iterdir():
        if (f.is_file()
                and f.suffix.lower() in VALID_EXTENSIONS
                and f.stat().st_size >= MIN_FILE_SIZE
                and not _is_ignored(f.name)):
            result.append(f)
    return sorted(result)


def _build_file_record(path: Path, process_pipeline: list) -> FileRecord:
    h = hashlib.sha256()
    try:
        with open(path, "rb") as fh:
            h.update(fh.read(16384))
    except Exception:
        pass
    fr = FileRecord(
        filename=path.name,
        path=str(path),
        extension=path.suffix.lower(),
        file_size_bytes=path.stat().st_size,
        file_hash=h.hexdigest()[:16],
        format_class=path.suffix.lower().lstrip("."),
    )
    for entry in process_pipeline:
        if entry.get("output") == path.name:
            fr.source_type         = "generated"
            fr.generated_from      = entry.get("generated_from", [])
            fr.tool                = entry.get("tool", "")
            fr.process             = entry.get("process", "")
            fr.lineage_confidence  = entry.get("confidence", 0.9)
            break
    return fr


def _infer_format(files: list) -> str:
    exts = {f.suffix.lower() for f in files}
    if ".png"  in exts: return "PNG"
    if ".jpg"  in exts: return "JPG"
    if ".jpeg" in exts: return "JPG"
    if ".tif"  in exts: return "TIFF"
    if ".tiff" in exts: return "TIFF"
    if ".exr"  in exts: return "EXR"
    return "MIX"


def _infer_resolution(files: list, folder: Path) -> str:
    for f in files:
        m = re.search(r'[_\-](\d{1,2}[kK])[_\-\.]', f.stem + ".", re.IGNORECASE)
        if m:
            return m.group(1).upper()
        m = re.search(r'[_\-](4096|8192|2048|1024|512)[_\-\.]', f.stem + ".")
        if m:
            px = int(m.group(1))
            for thr, lbl in ((7681,"8K"),(3841,"4K"),(2049,"2K"),(1025,"1K")):
                if px >= thr: return lbl
    try:
        from PIL import Image
        with Image.open(files[0]) as img:
            px = max(img.size)
            for thr, lbl in ((7681,"8K"),(3841,"4K"),(2049,"2K"),(1025,"1K")):
                if px >= thr: return lbl
    except Exception:
        pass
    return "UNKNOWN"


# ── Variant Building ──────────────────────────────────────────────────────────

def _build_variants(folder: Path, dva_id: str, process_pipeline: list) -> list:
    """
    Detect and build Variant objects for a material folder.
    Priority: named variants > resolution subfolders > flat.
    """
    variants = []
    named_variants = []
    resolution_folders = []

    for child in sorted(folder.iterdir()):
        if not child.is_dir():
            continue
        kind, res, fmt = _classify_subfolder(child.name)
        if kind == "variant_named":
            named_variants.append((child, res, fmt))
        elif kind == "resolution":
            resolution_folders.append((child, res))

    # ── Case 1: AmbientCG named variants ─────────────────────────────────────
    if named_variants:
        for vfolder, res, fmt in named_variants:
            files = _collect_image_files(vfolder)
            if not files:
                continue
            v = Variant(
                variant_id=f"{dva_id}_{res}_{fmt}",
                resolution=res,
                format=fmt,
            )
            for fp in files:
                v.files.append(_build_file_record(fp, process_pipeline))
            variants.append(v)
        return variants

    # ── Case 2: Resolution subfolders ────────────────────────────────────────
    if resolution_folders:
        for res_folder, res in resolution_folders:
            fmt_subfolders = [
                c for c in res_folder.iterdir()
                if c.is_dir() and _classify_subfolder(c.name)[0] == "format"
            ]
            if fmt_subfolders:
                for fmt_folder in sorted(fmt_subfolders):
                    _, _, fmt = _classify_subfolder(fmt_folder.name)
                    files = _collect_image_files(fmt_folder)
                    if not files:
                        continue
                    v = Variant(
                        variant_id=f"{dva_id}_{res}_{fmt or 'UNK'}",
                        resolution=res,
                        format=fmt,
                    )
                    for fp in files:
                        v.files.append(_build_file_record(fp, process_pipeline))
                    variants.append(v)
            else:
                files = _collect_image_files(res_folder)
                if not files:
                    continue
                fmt = _infer_format(files)
                v = Variant(
                    variant_id=f"{dva_id}_{res}_{fmt}",
                    resolution=res,
                    format=fmt,
                )
                for fp in files:
                    v.files.append(_build_file_record(fp, process_pipeline))
                variants.append(v)
        return variants

    # ── Case 3: Flat structure ────────────────────────────────────────────────
    direct = _collect_image_files(folder)
    if direct:
        res = _infer_resolution(direct, folder)
        fmt = _infer_format(direct)
        v = Variant(
            variant_id=f"{dva_id}_{res}_{fmt}",
            resolution=res,
            format=fmt,
        )
        for fp in direct:
            v.files.append(_build_file_record(fp, process_pipeline))
        variants.append(v)

    return variants


# ── Sidecar Loaders ───────────────────────────────────────────────────────────

def _load_material_info(folder: Path, mat: MaterialSet):
    p = folder / "material_info.json"
    if not p.exists():
        return
    try:
        data = json.loads(p.read_text(encoding="utf-8"))
        if "material" in data:
            mat.material_name = re.sub(r'\s+', '_', data["material"].strip())
        mat.category        = data.get("category", mat.category)
        mat.scale           = data.get("scale", "")
        mat.source          = data.get("source", mat.source)
        mat.custom_variants = data.get("variants", [])
        mat.notes           = data.get("notes", "")
    except Exception as e:
        logger.warning(f"material_info.json error ({folder.name}): {e}")


def _load_process_json(folder: Path, mat: MaterialSet):
    p = folder / "process.json"
    if not p.exists():
        return
    try:
        data = json.loads(p.read_text(encoding="utf-8"))
        mat.has_process_data = True
        mat.process_pipeline = data.get("pipeline", [])
        if "source" in data:
            mat.source = data["source"]
    except Exception as e:
        logger.warning(f"process.json error ({folder.name}): {e}")


# ── Ingestor ──────────────────────────────────────────────────────────────────

class Ingestor:
    """
    Scans 01_RAW_ARCHIVE.
    Returns one MaterialSet per physical material, variants properly grouped.

    Usage:
        result = Ingestor("./01_RAW_ARCHIVE").scan()
    """

    def __init__(self, root_path: str, min_file_size: int = MIN_FILE_SIZE):
        self.root = Path(root_path)
        self.min_file_size = min_file_size
        self._materials: list = []

    def scan(self) -> dict:
        if not self.root.exists():
            raise FileNotFoundError(f"Root not found: {self.root}")
        logger.info(f"Ingestor v2.1 scanning: {self.root}")
        self._materials = []

        material_folders = self._discover_material_folders()
        logger.info(f"Ingestor: {len(material_folders)} material folder(s)")

        for folder in sorted(material_folders):
            mat = self._process_folder(folder)
            if mat:
                self._materials.append(mat)
                logger.info(f"  [{mat.provider_code}] {mat.material_name} → {len(mat.variants)} variant(s)")

        logger.info(f"Ingestor: {len(self._materials)} MaterialSet(s) built")
        return self._build_output()

    def get_materials(self) -> list:
        return self._materials

    def save_manifest(self, output_path: str) -> str:
        out = self._build_output()
        p = Path(output_path)
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text(json.dumps(out, indent=2, ensure_ascii=False))
        logger.info(f"Manifest saved → {output_path}")
        return output_path

    def summary(self) -> dict:
        return self._build_output()["summary"]

    # ── Discovery ─────────────────────────────────────────────────────────────

    def _discover_material_folders(self) -> list:
        """
        Walk root BFS. Identify material folders.
        Claim variant subfolders so they are not treated as separate materials.
        """
        candidates = []
        claimed: set = set()

        queue = [self.root]
        while queue:
            current = queue.pop(0)

            # Skip claimed folders (they are variants owned by a parent)
            if current in claimed:
                continue

            # Skip provider wrappers at depth 1
            if current != self.root and _is_provider_folder(current, self.root):
                # Enqueue children (material folders inside provider)
                for child in sorted(current.iterdir()):
                    if child.is_dir():
                        queue.append(child)
                continue

            # Skip folders that look like variants themselves
            kind, _, _ = _classify_subfolder(current.name)
            if kind in ("variant_named", "resolution", "format") and current != self.root:
                continue

            # Check if this is a material
            if current != self.root and _is_material_folder(current):
                candidates.append(current)
                # Claim all direct and grandchild dirs (variant subfolders)
                for child in current.iterdir():
                    if child.is_dir():
                        claimed.add(child)
                        for grandchild in child.iterdir():
                            if grandchild.is_dir():
                                claimed.add(grandchild)
                continue

            # Not a material yet — keep walking
            for child in sorted(current.iterdir()):
                if child.is_dir() and child not in claimed:
                    queue.append(child)

        return candidates

    # ── Material Processing ───────────────────────────────────────────────────

    def _process_folder(self, folder: Path) -> Optional[MaterialSet]:
        provider, provider_code = _detect_provider(folder, self.root)
        raw_name  = folder.name.strip()
        mat_name  = re.sub(r'\s+', '_', raw_name)

        uid    = str(uuid.uuid5(uuid.NAMESPACE_DNS, str(folder.resolve())))
        dva_id = f"DVA_{provider_code}_{mat_name}"

        mat = MaterialSet(
            id=uid,
            dva_id=dva_id,
            provider=provider,
            provider_code=provider_code,
            material_name=mat_name,
            folder_path=str(folder),
            source=provider.lower(),
        )

        _load_material_info(folder, mat)
        _load_process_json(folder, mat)

        mat.variants = _build_variants(folder, dva_id, mat.process_pipeline)

        if not mat.variants:
            return None

        # Collect non-image files
        for child in folder.rglob("*"):
            if child.is_file():
                ext = child.suffix.lower()
                if ext in ASSET_EXTENSIONS:
                    mat.asset_files.append(str(child))
                elif ext not in VALID_EXTENSIONS and not _is_ignored(child.name):
                    mat.extra_files.append(str(child))

        return mat

    # ── Output ────────────────────────────────────────────────────────────────

    def _build_output(self) -> dict:
        mats = [m.to_dict() for m in self._materials]
        total_files = sum(len(v["files"]) for m in mats for v in m["variants"])
        return {
            "materials": mats,
            "summary": {
                "total_materials":   len(mats),
                "total_files":       total_files,
                "providers":         sorted({m["provider"] for m in mats}),
                "resolutions_found": sorted({v["resolution"] for m in mats for v in m["variants"]}),
            },
        }

# ── Result Dataclass ──────────────────────────────────────────────────────────

@dataclass
class NamingResult:
    map_type: str           # "albedo", "normal", "roughness", … or "unknown"
    confidence: float       # 0.0 – 1.0
    method: str             # "exact_token" | "suffix_match" | "keyword_match" | "unknown"
    is_packed: bool = False
    packed_channels: dict = field(default_factory=dict)
    convention: Optional[str] = None  # "DX" | "GL" | None (for normal maps)
    color_space: str = ""

    def to_dict(self) -> dict:
        return {
            "map_type":        self.map_type,
            "confidence":      self.confidence,
            "method":          self.method,
            "is_packed":       self.is_packed,
            "packed_channels": self.packed_channels,
            "convention":      self.convention,
            "color_space":     self.color_space,
        }


# ── Core Classifier ───────────────────────────────────────────────────────────

class NamingIntelligence:
    """
    Classifies PBR texture filenames into map types.

    Three detection passes (in priority order):
    1. Exact token match — highest confidence (0.97)
       Splits filename into tokens and checks each against exact_tokens list.
    2. Suffix match — high confidence (0.90)
       Checks if lowercase stem ends with a known suffix.
    3. Keyword match — medium confidence (0.75)
       Substring search across the lowercase stem.
    """

    def __init__(self, patterns: Optional[dict] = None):
        self.patterns = patterns or MAP_PATTERNS

    # ── Public API ────────────────────────────────────────────────────────────

    def classify(self, filename: str) -> NamingResult:
        """
        Classify a single filename.

        Args:
            filename: bare filename or full path (only stem is used)

        Returns:
            NamingResult with map_type, confidence, method, etc.
        """
        from pathlib import Path as _Path
        stem = _Path(filename).stem

        result = self._run_passes(stem)

        # Detect normal map convention (DX vs GL)
        if result.map_type == "normal":
            result.convention = self._detect_convention(stem)

        return result

    def classify_batch(self, file_list: list) -> dict:
        """
        Classify a list of filenames.

        Returns:
            {filename: NamingResult, ...}
        """
        return {f: self.classify(f) for f in file_list}

    def summarize_set(self, results: dict) -> dict:
        """
        Given a classify_batch result, produce a coverage summary.

        Returns:
            {
              "covered": ["albedo", "normal", ...],
              "missing": ["metallic", ...],
              "unknown": ["file_x.png", ...],
              "packed": ["Rock_ORM.png", ...],
              "normal_conventions": {"DX": [...], "GL": [...]},
            }
        """
        covered: set = set()
        unknown: list = []
        packed: list = []
        normal_conventions: dict = {"DX": [], "GL": [], "ambiguous": []}

        required = {"albedo", "normal", "roughness"}

        for fname, r in results.items():
            if r.map_type == "unknown":
                unknown.append(fname)
            else:
                if r.is_packed:
                    packed.append(fname)
                    # Packed maps cover their constituent channels
                    for ch in r.packed_channels.values():
                        covered.add(ch.replace("_inv", ""))
                else:
                    covered.add(r.map_type)

                if r.map_type == "normal":
                    key = r.convention if r.convention else "ambiguous"
                    normal_conventions[key].append(fname)

        missing = sorted(required - covered)

        return {
            "covered":            sorted(covered),
            "missing":            missing,
            "unknown_files":      unknown,
            "packed_maps":        packed,
            "normal_conventions": normal_conventions,
        }

    # ── Detection Passes ─────────────────────────────────────────────────────

    def _run_passes(self, stem: str) -> NamingResult:
        lower = stem.lower()
        tokens = set(re.split(r'[_\-\s]+', lower))

        # Pass 1 — exact token match (highest confidence)
        for map_type, cfg in self.patterns.items():
            for tok in cfg.get("exact_tokens", []):
                if tok in tokens:
                    return NamingResult(
                        map_type=map_type,
                        confidence=0.97,
                        method="exact_token",
                        is_packed=cfg.get("packed", False),
                        packed_channels=cfg.get("channels", {}),
                        color_space=cfg.get("color_space", ""),
                    )

        # Pass 2 — suffix match
        for map_type, cfg in self.patterns.items():
            for suffix in cfg.get("suffixes", []):
                if lower.endswith(suffix):
                    return NamingResult(
                        map_type=map_type,
                        confidence=0.90,
                        method="suffix_match",
                        is_packed=cfg.get("packed", False),
                        packed_channels=cfg.get("channels", {}),
                        color_space=cfg.get("color_space", ""),
                    )

        # Pass 3 — keyword substring match
        for map_type, cfg in self.patterns.items():
            for kw in cfg.get("keywords", []):
                if kw in lower:
                    return NamingResult(
                        map_type=map_type,
                        confidence=0.75,
                        method="keyword_match",
                        is_packed=cfg.get("packed", False),
                        packed_channels=cfg.get("channels", {}),
                        color_space=cfg.get("color_space", ""),
                    )

        return NamingResult(
            map_type="unknown",
            confidence=0.0,
            method="unknown",
        )

    def _detect_convention(self, stem: str) -> Optional[str]:
        if _NORMAL_CONVENTION_RE.search(stem):
            return "DX"
        if _NORMAL_GL_RE.search(stem):
            return "GL"
        # Heuristic: files from UE5/Quixel pipelines default to DX
        if re.search(r'quixel|ue5|unreal|megascans', stem, re.IGNORECASE):
            return "DX"
        return None  # ambiguous


# ── Convenience Functions ─────────────────────────────────────────────────────

def detect_map_type(filename: str) -> dict:
    """
    Convenience wrapper around NamingIntelligence.classify().

    Returns dict compatible with pipeline JSON output:
        {"type": "albedo", "confidence": 0.97, "method": "exact_token"}
    """
    ni = NamingIntelligence()
    r = ni.classify(filename)
    return {"type": r.map_type, "confidence": r.confidence, "method": r.method}


def analyze_files(file_list: list) -> dict:
    """
    Batch classify a list of filenames.

    Returns:
        {filename: {"type": ..., "confidence": ..., "method": ..., ...}, ...}
    """
    ni = NamingIntelligence()
    results = ni.classify_batch(file_list)
    return {fname: r.to_dict() for fname, r in results.items()}

Differenze tra DeepBump e normalmap-generator per mappe normal
In sintesi: DeepBump è un tool ML “intelligente” che ricostruisce normali (e height) direttamente da una foto/albedo, mentre normalmap-generator è uno script deterministico che converte una height map già esistente in normal map.^3_1^3_3
---
Differenza di Input e Output
DeepBump
Input: tipicamente una color image / albedo singola, o una normal, direttamente dal nodo immagine in Blender o via CLI.^3_2
Output:
Normal map da singola foto.
Height map a partire dalla normal (tramite modulo NormalHeight integrato).^3_4^3_1
Curvature map derivata dalla normal (utile anche per maschere).^3_1
normalmap-generator
Input: height map in scala di grigi (deve già rappresentare l’altezza).^3_6
Output: solo normal map derivata da quella height (nessuna height “intelligente” da foto).^3_3
---
Approccio algoritmico
DeepBump
Usa una rete neurale encoder–decoder tipo U-Net con backbone MobileNetV2 per ricostruire le normali da una singola immagine, addestrata su dataset fotogrammetry + materiali procedurali.^3_2
L’idea è superare il metodo “grayscale = height” citato nella doc, che viene considerato una approssimazione grezza. DeepBump cerca pattern strutturali nella texture (rilievi, cavità) anche quando il contrasto luminanza non corrisponde direttamente alla geometria.^3_2
normalmap-generator
È uno script classico “height → normal”: calcola la normale come gradiente (derivate X/Y) della height map, usando un kernel (Sobel/central difference) e produce il vettore normalizzato per ogni pixel.^3_6
Nessun ML, nessuna inferenza semantica: se la height è sbagliata o ambigua, la normal sarà altrettanto sbagliata.

def parse_process_description(text: str, output_filename: str = "") -> dict:
    """
    Convert free-text process description into structured JSON.

    Examples:
        "normal da albedo con materialize"
        → {"process": "normal_from_albedo", "tool": "Materialize",
           "generated_from": ["albedo"], "output_type": "normal"}

        "roughness generata da gloss invertita in photoshop"
        → {"process": "roughness_from_gloss", "tool": "Photoshop",
           "generated_from": ["gloss"]}
    """
    lower = text.lower().strip()

    result = {
        "process":        "unknown",
        "tool":           "",
        "generated_from": [],
        "output_type":    "",
        "confidence":     0.5,
        "raw_description": text,
    }

    # Detect tool
    for kw, tool_name in _TOOL_KEYWORDS.items():
        if kw in lower:
            result["tool"] = tool_name
            break

    # Detect process type from compound keywords
    for phrase, proc_data in _PROCESS_MAP.items():
        if phrase in lower:
            result["process"] = proc_data["process"]
            if proc_data["generated_from"]:
                result["generated_from"] = proc_data["generated_from"]
            if not result["tool"] and proc_data["tool"]:
                result["tool"] = proc_data["tool"]
            result["confidence"] = 0.85
            break

    # If no compound match, try to infer from individual map type tokens
    if result["process"] == "unknown":
        found_types = []
        for kw, mt in _MAP_TYPE_KEYWORDS.items():
            if kw in lower:
                found_types.append(mt)
        if len(found_types) >= 2:
            # Two map types mentioned → likely "X from Y" relationship
            result["process"]        = f"{found_types[0]}_from_{found_types[1]}"
            result["generated_from"] = [found_types[1]]
            result["output_type"]    = found_types[0]
            result["confidence"]     = 0.65
        elif len(found_types) == 1:
            result["output_type"]    = found_types[0]
            result["confidence"]     = 0.55

    # Infer output type from filename if provided
    if not result["output_type"] and output_filename:
        for kw, mt in _MAP_TYPE_KEYWORDS.items():
            if kw in output_filename.lower():
                result["output_type"] = mt
                break

    return result

                        ".blend", ".sbsar", ".uasset", ".usdc", ".mtlx"}

    def __init__(
        self,
        raw_root:    str = "01_RAW_ARCHIVE",
        custom_root: str = "02_CUSTOM",
    ):
        self.raw_root    = Path(raw_root)
        self.custom_root = Path(custom_root)

    # ── Public API ────────────────────────────────────────────────────────────

    def import_files(
        self,
        source:              str,
        asset_type:          str,           # "RAW" or "CUSTOM"
        provider:            str,
        material_name:       str,
        process_description: str = "",
        notes:               str = "",
        dry_run:             bool = False,
    ) -> dict:
        """
        Import files from source into the correct archive folder.

        Args:
            source:              path to file or folder to import
            asset_type:          "RAW" → 01_RAW_ARCHIVE, "CUSTOM" → 02_CUSTOM
            provider:            e.g. "AmbientCG", "Quixel", "Custom"
            material_name:       e.g. "Tiles_201", "Rock_Granite"
            process_description: free-text description of how maps were generated
            notes:               any extra notes
            dry_run:             if True, only report what would happen

        Returns:
            dict with result summary
        """
        source_path = Path(source)
        if not source_path.exists():
            raise FileNotFoundError(f"Source not found: {source}")

        asset_type = asset_type.upper()
        if asset_type not in ("RAW", "CUSTOM"):
            raise ValueError("asset_type must be 'RAW' or 'CUSTOM'")

        # Determine destination
        root      = self.raw_root if asset_type == "RAW" else self.custom_root
        mat_clean = re.sub(r'[\s]+', '_', material_name.strip())
        dest_dir  = root / provider / mat_clean
        dest_dir.mkdir(parents=True, exist_ok=True)

        # Collect files to copy
        files_to_copy = self._collect_files(source_path)
        logger.info("ImportAssistant: %d file(s) to import → %s", len(files_to_copy), dest_dir)

        if dry_run:
            return {
                "status":     "dry_run",
                "destination": str(dest_dir),
                "files":       [str(f) for f in files_to_copy],
            }

        # Copy files
        copied = []
        for f in files_to_copy:
            dest_file = dest_dir / f.name
            shutil.copy2(f, dest_file)
            copied.append(str(dest_file))
            logger.debug("  copied: %s", f.name)

        # Write material_info.json
        self._write_material_info(dest_dir, mat_clean, provider, asset_type, notes)

        # Write process.json if description given
        if process_description:
            self._write_process_json(dest_dir, mat_clean, process_description, files_to_copy)

        return {
            "status":      "ok",
            "destination": str(dest_dir),
            "files_copied": len(copied),
            "files":       copied,
            "sidecars":    ["material_info.json"] + (["process.json"] if process_description else []),
        }

    # ── Sidecar Writers ───────────────────────────────────────────────────────

    def _write_material_info(
        self,
        dest_dir:      Path,
        material_name: str,
        provider:      str,
        asset_type:    str,
        notes:         str,
    ):
        info = {
            "material":    material_name,
            "provider":    provider,
            "type":        asset_type,
            "import_date": datetime.now().isoformat(),
            "source":      asset_type.lower(),
            "notes":       notes,
        }
        (dest_dir / "material_info.json").write_text(
            json.dumps(info, indent=2, ensure_ascii=False),
            encoding="utf-8",
        )

    def _write_process_json(
        self,
        dest_dir:    Path,
        mat_name:    str,
        description: str,
        files:       list,
    ):
        pipeline_entries = []

        # Build one entry per file that appears to be generated
        for f in files:
            parsed = parse_process_description(description, f.name)
            if parsed["process"] != "unknown":
                entry = {
                    "output":         f.name,
                    "type":           parsed.get("output_type", "unknown"),
                    "process":        parsed["process"],
                    "tool":           parsed["tool"],
                    "generated_from": parsed["generated_from"],
                    "confidence":     parsed["confidence"],
                    "description":    description,
                }
                pipeline_entries.append(entry)
                break   # one representative entry (user can edit manually)

        process = {
            "material": mat_name,
            "source":   "custom",
            "pipeline": pipeline_entries,
            "raw_description": description,
        }
        (dest_dir / "process.json").write_text(
            json.dumps(process, indent=2, ensure_ascii=False),
            encoding="utf-8",
        )

    # ── Helpers ───────────────────────────────────────────────────────────────

    def _collect_files(self, source: Path) -> list:
        """Return list of valid files from source (file or directory)."""
        if source.is_file():
            return [source] if source.suffix.lower() in self.VALID_EXTENSIONS else []

        result = []
        for f in source.rglob("*"):
            if f.is_file() and f.suffix.lower() in self.VALID_EXTENSIONS:
                result.append(f)
        return sorted(result)

def _prompt(question: str, default: str = "") -> str:
    if default:
        val = input(f"  {question} [{default}]: ").strip()
        return val if val else default
    return input(f"  {question}: ").strip()

    """Interactive CLI for manual import."""

    dry_run_str = _prompt("Dry run? (y/n)", "n")

    try:
        result = ia.import_files(

            asset_type=asset_type,
            provider=provider,
            material_name=mat_name,
            process_description=description,
            notes=notes,
            dry_run=dry_run,
        )
        print()
        if result["status"] == "dry_run":
            print(f"  [DRY RUN] Would import {len(result['files'])} file(s) → {result['destination']}")
            for f in result["files"]:
                print(f"    {Path(f).name}")
        else:
            print(f"  [OK] {result['files_copied']} file(s) imported → {result['destination']}")
            print(f"  Sidecars: {', '.join(result['sidecars'])}")
            if description:

                print(f"    generated_from: {parsed['generated_from']}")
                print(f"    confidence:     {parsed['confidence']:.0%}")
    except Exception as e:
        print(f"\n  [ERROR] {e}")

    print()


if __name__ == "__main__":
