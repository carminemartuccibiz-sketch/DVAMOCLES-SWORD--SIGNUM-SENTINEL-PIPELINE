import os
import json
import shutil
import subprocess
from pathlib import Path

# ─────────────────────────────
# CONFIG
# ─────────────────────────────

RAW_DIR = "01_RAW_ARCHIVE"
CUSTOM_DIR = "02_CUSTOM"

OLLAMA_MODEL = "mistral"  # cambia se vuoi (llama3, mixtral ecc)

VALID_EXT = [".png", ".jpg", ".jpeg", ".tif", ".tiff", ".exr"]

# ─────────────────────────────
# OLLAMA CALL
# ─────────────────────────────

def ask_ai(prompt):
    try:
        result = subprocess.run(
            ["ollama", "run", OLLAMA_MODEL, prompt],
            capture_output=True,
            text=True
        )
        return result.stdout.strip()
    except Exception as e:
        return None

# ─────────────────────────────
# USER INPUT (fallback)
# ─────────────────────────────

def ask_user(question):
    return input(f"{question}: ").strip()

# ─────────────────────────────
# CLASSIFICATION
# ─────────────────────────────

def classify_files(file_list):
    """
    Decide:
    - RAW o CUSTOM
    - materiale
    - provider
    """

    filenames = [Path(f).name for f in file_list]

    prompt = f"""
You are a texture pipeline assistant.

Given these files:
{filenames}

Return JSON:

{{
  "type": "raw or custom",
  "material": "material_name",
  "provider": "provider_name",
  "notes": "optional"
}}

ONLY JSON.
"""

    response = ask_ai(prompt)

    try:
        return json.loads(response)
    except:
        return None

# ─────────────────────────────
# PROCESS DESCRIPTION → JSON
# ─────────────────────────────

def generate_process_json(description, files):
    prompt = f"""
Convert this description into structured JSON pipeline.

Description:
{description}

Files:
{files}

Return JSON:
{{
  "pipeline": [
    {{
      "step": "...",
      "tool": "...",
      "input": "...",
      "output": "...",
      "confidence": 0.9
    }}
  ]
}}
ONLY JSON.
"""

    response = ask_ai(prompt)

    try:
        return json.loads(response)
    except:
        return {"pipeline": []}

# ─────────────────────────────
# MAIN IMPORT
# ─────────────────────────────

def import_files(input_folder):

    input_path = Path(input_folder)
    files = [str(f) for f in input_path.glob("*") if f.suffix.lower() in VALID_EXT]

    if not files:
        print("No valid files found")
        return

    print(f"\nFound {len(files)} files")

    ai_data = classify_files(files)

    # fallback manuale
    if not ai_data:
        print("AI failed → manual mode")
        type_ = ask_user("RAW or CUSTOM?")
        material = ask_user("Material name?")
        provider = ask_user("Provider?")
    else:
        print("\nAI RESULT:")
        print(json.dumps(ai_data, indent=2))

        confirm = ask_user("Confirm? (y/n)")
        if confirm.lower() != "y":
            type_ = ask_user("RAW or CUSTOM?")
            material = ask_user("Material name?")
            provider = ask_user("Provider?")
        else:
            type_ = ai_data["type"]
            material = ai_data["material"]
            provider = ai_data["provider"]

    # ───────────────
    # DESTINATION
    # ───────────────

    base_dir = RAW_DIR if type_ == "raw" else CUSTOM_DIR

    dest = Path(base_dir) / provider / material
    dest.mkdir(parents=True, exist_ok=True)

    # copy files
    for f in files:
        shutil.copy2(f, dest)

    print(f"\nMoved to: {dest}")

    # ───────────────
    # CUSTOM PROCESS
    # ───────────────

    if type_ == "custom":
        desc = ask_user("Describe process (optional)")
        if desc:
            process_data = generate_process_json(desc, files)

            with open(dest / "process.json", "w") as f:
                json.dump(process_data, f, indent=2)

            print("process.json created")

# ─────────────────────────────
# CLI
# ─────────────────────────────

if __name__ == "__main__":
    folder = input("Folder to import: ").strip()
    import_files(folder)
